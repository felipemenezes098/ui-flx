import { Check } from "lucide-react";
import Balancer from "react-wrap-balancer";
import { createImageUrlBuilder } from "@sanity/image-url";

import { Cta } from "@/components/flx/sanity/shared/cta";
import { Blocks } from "@/sanity.types";
import { client } from "@/sanity/client";

type Hero06Block = Extract<Blocks[number], { _type: "hero06" }>;

interface Hero06Props {
  block: Hero06Block;
}

const builder = createImageUrlBuilder(client);

export function Hero06({ block }: Readonly<Hero06Props>) {
  const previewUrl = block.preview?.asset
    ? builder.image(block.preview).url()
    : null;

  return (
    <section className="bg-background relative w-full overflow-hidden">
      <div className="relative mx-auto flex max-w-5xl flex-col items-center px-6 pt-16 pb-16 text-center sm:pt-24">
        <h1 className="text-foreground max-w-3xl text-4xl font-semibold tracking-tight sm:text-6xl">
          <Balancer>
            {block.title}
            {block.highlight && (
              <>
                {" "}
                <span className="text-muted-foreground">{block.highlight}</span>
              </>
            )}
          </Balancer>
        </h1>

        {block.description && (
          <p className="text-muted-foreground mt-6 max-w-xl text-base sm:text-lg">
            <Balancer>{block.description}</Balancer>
          </p>
        )}

        <div className="mt-8 flex w-full flex-col items-center justify-center gap-3 sm:w-auto sm:flex-row">
          {block.primaryCTA && (
            <Cta cta={block.primaryCTA} className="w-full px-6 sm:w-fit" />
          )}
          {block.secondaryCTA && (
            <Cta cta={block.secondaryCTA} className="w-full px-6 sm:w-fit" />
          )}
        </div>

        {block.highlights && block.highlights.length > 0 && (
          <ul className="text-muted-foreground mt-6 flex flex-wrap items-center justify-center gap-x-6 gap-y-2 text-sm">
            {block.highlights.map((item) => (
              <li key={item} className="inline-flex items-center gap-1.5">
                <Check className="text-foreground/60 size-4" />
                {item}
              </li>
            ))}
          </ul>
        )}

        {previewUrl && (
          <div className="relative mt-16 w-full">
            <div className="border-border bg-card overflow-hidden rounded-2xl border shadow-sm">
              <img
                src={previewUrl}
                alt={block.title ?? "Preview"}
                loading="lazy"
                decoding="async"
                className="aspect-[16/10] w-full object-cover"
              />
            </div>
            <div className="from-background pointer-events-none absolute inset-x-0 -bottom-px h-24 bg-gradient-to-t to-transparent" />
          </div>
        )}

        {block.logos && block.logos.length > 0 && (
          <div className="border-border/60 mt-4 w-full border-t pt-10">
            {block.logosLabel && (
              <p className="text-muted-foreground text-xs tracking-wide uppercase">
                {block.logosLabel}
              </p>
            )}
            <div className="mt-6 flex flex-wrap items-center justify-center gap-x-10 gap-y-4">
              {block.logos.map((logo) => (
                <span
                  key={logo}
                  className="text-muted-foreground/70 text-lg font-semibold tracking-tight"
                >
                  {logo}
                </span>
              ))}
            </div>
          </div>
        )}
      </div>
    </section>
  );
}
