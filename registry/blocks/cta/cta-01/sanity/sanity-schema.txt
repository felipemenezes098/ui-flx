import {defineField, defineType} from 'sanity'

export const cta01Block = defineType({
  name: 'cta01',
  title: 'Call to Action 01',
  type: 'object',
  groups: [
    {
      name: 'content',
      title: 'Content',
    },
  ],
  fields: [
    defineField({
      name: 'title',
      title: 'Title',
      type: 'string',
      group: 'content',
    }),
    defineField({
      name: 'description',
      title: 'Description',
      type: 'text',
      group: 'content',
    }),
    defineField({
      name: 'cta',
      title: 'CTA',
      type: 'cta',
      group: 'content',
    }),
  ],
  preview: {
    select: {
      title: 'title',
      subtitle: 'description',
    },
    prepare(selection) {
      const {title, subtitle} = selection

      return {
        title: title ? `Call to Action 01 - ${title}` : 'Call to Action 01',
        subtitle: subtitle ?? '',
      }
    },
  },
})
