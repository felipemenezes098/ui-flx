"use client";

import { createImageUrlBuilder } from "@sanity/image-url";
import { icons } from "lucide-react";
import Balancer from "react-wrap-balancer";
import { Icon } from "../shared/dynamic-icon";
import { Blocks } from "@/sanity.types";
import { client } from "@/sanity/client";
import { cn } from "@/lib/utils";

type Content02Block = Extract<
  Blocks[number],
  { _type: "content02" }
>;

interface Content02Props {
  block: Content02Block;
}

const builder = createImageUrlBuilder(client);

export function Content02({ block }: Readonly<Content02Props>) {
  const title = block.title;
  const description = block.description;
  const items = block.items ?? [];

  if (items.length === 0) {
    return null;
  }

  return (
    <section className="w-full">
      <div className="mx-auto flex w-full max-w-6xl flex-col gap-10 px-6 py-12 sm:gap-12 sm:py-16">
      {(title || description) && (
        <div className="flex max-w-2xl flex-col gap-3">
          {title && (
            <h2 className="text-foreground font-serif text-2xl font-normal tracking-tight text-balance sm:text-3xl">
              <Balancer>{title}</Balancer>
            </h2>
          )}
          {description && (
            <p className="text-muted-foreground max-w-xl text-sm sm:text-base">
              <Balancer>{description}</Balancer>
            </p>
          )}
        </div>
      )}
      <div className="grid grid-cols-1 gap-8 sm:gap-10 md:grid-cols-3">
        {items.map((item, index) => {
          const imageUrl = item.image?.asset
            ? builder.image(item.image).url()
            : null;

          return (
            <article key={item._key ?? index} className="group/card flex flex-col gap-4">
              <div className="group/image relative aspect-[4/5] overflow-hidden rounded-lg outline outline-black/10 dark:outline-white/10">
                {imageUrl && (
                  <img
                    src={imageUrl}
                    alt={item.image?.alt ?? item.title ?? "Content 02"}
                    loading={index === 0 ? "eager" : "lazy"}
                    decoding="async"
                    className="size-full object-cover transition-transform duration-500 ease-[cubic-bezier(0.22,1,0.36,1)] group-hover/image:scale-[1.03]"
                  />
                )}
              </div>
              <div className="flex flex-col gap-1.5">
                <div className="flex items-center gap-2">
                  {item.icon && (
                    <Icon
                      name={item.icon as keyof typeof icons}
                      className="text-muted-foreground size-4 shrink-0"
                    />
                  )}
                  <h3 className="text-base font-medium tracking-tight">
                    {item.title}
                  </h3>
                </div>
                {item.description && (
                  <p className="text-muted-foreground text-sm">
                    {item.description}
                  </p>
                )}
              </div>
            </article>
          );
        })}
      </div>
      </div>
    </section>
  );
}
